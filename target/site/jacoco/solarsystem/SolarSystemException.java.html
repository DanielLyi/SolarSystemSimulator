package solarsystem;

/**
 * Solar System exception.
 * @author Nico Kuijpers
 */
public class SolarSystemException extends Exception {

    /**
     * Creates a new instance of SolarSystemException without
     * detail message.
     */
    public SolarSystemException() {
        super();
    }

    /**
     * Creates an instance of SolarSystemException with the
     * specified detail message.
     * @param message the detail message.
     */
    public SolarSystemException(String message) {
        super(message);
    }
}
